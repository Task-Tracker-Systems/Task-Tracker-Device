58template <typename CharType>
59std::vector<std::basic_string<CharType>> tokenizeQuoted(const std::basic_string<CharType> &input)
60{
61 std::basic_stringstream<CharType> iss(input);
62 std::vector<std::basic_string<CharType>> tokens;
63
64 std::basic_string<CharType> token;
65 while (!iss.eof())
66 {
67 iss >> std::quoted(token);
68 if (iss.fail())
69 {
70 throw std::runtime_error("failed to tokenize string: '" + input + "'");
71 }
72 tokens.push_back(token);
73 }
74
75 return tokens;

96 template <class ArgumentType>
97 static ArgumentType extractArgument(const Option<ArgumentType, CharT> &option,
98 std::vector<std::basic_string<CharT>> &args)
99 {
100 const auto itAllOptions = std::begin(args); // Skip the command name
101
102 // Find the first matching argument in the command line
103 const auto argIt = std::find_if(itAllOptions, std::end(args), [&option](const auto &arg) {
104 return option.doesMatchName(arg.c_str());
105 });
106
107 if (argIt != std::end(args))
108 {
109 // If a match is found, set the corresponding argument value
110 const auto itArgValueString = std::next(argIt);
111 const auto &argValueString = *itArgValueString;
112 std::basic_istringstream<CharT> iss(argValueString);
113 ArgumentType argument;
114 iss >> argument;
115 if (iss.fail())
116 {
117 throw std::runtime_error("argument to option " + std::basic_string<CharT>(option.labels[0]) + " could not be parsed: '" + argValueString + "'");
118 }
119 args.erase(argIt, std::next(itArgValueString));
120 return argument;
121 }
122 else
123 {
124 // If no match is found, set the default value
125 return option.defaultValue;
126 }
127 }
